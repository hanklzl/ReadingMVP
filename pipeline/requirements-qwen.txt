# Optional local TTS + forced-alignment dependencies for LMC_TTS_PROVIDER=qwen.
#
# These pull large model weights (several GB) on first use and are ONLY required
# when generating real audio with the open-source Qwen3 stack. Mock generation
# (LMC_TTS_PROVIDER=mock) and `python -m validator` do NOT need any of this.
#
# Install on the machine that runs the batch audio job:
#   pip install -r requirements.txt -r requirements-qwen.txt
#
# Models (downloaded lazily by from_pretrained / configurable via env):
#   - TTS:     Qwen/Qwen3-TTS-12Hz-1.7B-CustomVoice  (LMC_TTS_MODEL; warm storyteller
#              speaker "Serena" via LMC_TTS_VOICE — see README speaker list)
#   - Aligner: Qwen/Qwen3-ForcedAligner-0.6B         (LMC_TTS_ALIGNER_MODEL)
#
# PLATFORM CONSTRAINT (verified 2026-06): torch ships NO wheel for Intel macOS
# (macosx x86_64) past 2.2.2, and 2.2.2 only covers CPython 3.8-3.12. There is no
# torch wheel for CPython 3.13/3.14 on Intel macOS from PyPI, the PyTorch CPU
# index, or nightly. To run this stack on Intel macOS you must use Python 3.12
# (torch==2.2.2). On Apple Silicon / Linux any of 3.10-3.13 works.
#
# NOTE: qwen-tts pins transformers==4.57.3 and qwen-asr pins transformers==4.57.6;
# they cannot share one env unmodified. Run TTS and the forced-aligner in separate
# venvs, or relax one pin, when installing both.

qwen-tts>=0.1.1          # Qwen3TTSModel local text-to-speech (pulls transformers/torch/torchaudio)
qwen-asr>=0.0.6          # Qwen3ForcedAligner per-character timestamps (latest is 0.0.6, not 0.1.x)
torch>=2.2.0             # CPU is fine; on Intel macOS pin torch==2.2.2 + Python 3.12 (see note above)
soundfile>=0.12.1        # write generated waveforms to wav
